import { Playground, Props } from 'docz';
import { TypographyProvider } from '../Typography';
import LinkProvider from './LinkProvider';
import Link from './Link';

## **LinkProvider**

`<LinkProvider />` is responsible for passing down the theme to the `<Link />` component. It expects the theme to be of certain structure.



### **Example**

<Playground>
  {() => {
    const linkTheme = {
      linkTypes: {
        default: {
          normal: {
            color: '#1e88e5',
          },
          hover: {
            color: '#166dba',
          },
        },
        muted: {
          normal: {
            color: '#868e96',
          },
          hover: {
            color: '#6c747d',
          },
        },
      },
      common: {
        fontWeight: 'bold',
        fontFamily: `'Inter var', sans-serif`,
      },
    };

    const Example = () => (
      <TypographyProvider>
        <LinkProvider theme={linkTheme}>
          <>
            <Link className="margin-bottom">default link</Link>
            <Link linkType="muted">muted link</Link>
          </>
        </LinkProvider>
      </TypographyProvider>
    );

    return <Example />;
  }}
</Playground>

---



## **Link**

Link component grabs the theme given to `<LinkProvider />` and applies the style defined there based on the given `linkType` prop value.
