## **FormProvider**

`<FormProvider />` is responsible for passing down the theme to the all the form components. It expects the theme to be of certain structure.


### **Minimal example**

The minimal form theme has to include styles for valid and invalid states of form components.

<Playground>
  {() => {
    const formTheme = {
      // text input, textarea, and select
      regular: {
        states: {
          valid: {
            normal: {
              color: '#161616',
              borderColor: '#ced4da',
            },
          },
          invalid: {
            color: '#f4511e',
            borderColor: '#f4511e',
          },
        },
      },
      // inline input (i.e. no border)
      inline: {
        states: {
          valid: {
            normal: {
              color: '#161616',
            },
          },
          invalid: {
            color: '#F4511E',
            borderColor: '#F4511E',
          },
        },
      },
      // radio and checkbox
      misc: {
        states: {
          valid: {
            normal: {
              borderColor: '#ced4da',
            },
            checked: {
              backgroundColor: '#1E88E5',
            },
          },
          invalid: {
            borderColor: '#F4511E',
          },
        },
      },
      // toggle
      toggle: {
        states: {
          normal: {
            backgroundColor: '#ced4da',
          },
          checked: {
            backgroundColor: '#1E88E5',
          },
        },
      },
    };

    const Example = () => (
      /* TypographyProvider is required for FormProvider to work */
      <TypographyProvider>
        <FormProvider theme={formTheme}>
          <div className="max-width-m">
            <Input className="margin-bottom" />
            <InlineInput placeholder="Inline input" className="margin-bottom"/>
            <Textarea className="margin-bottom" />
            <Select className="margin-bottom">
              <option>option one</option>
              <option>option two</option>
            </Select>
            <Radio className="margin-bottom" />
            <Checkbox className="margin-bottom" />
            <div className="max-width-s">
              <Toggle onChange={() => {}} />
            </div>
          </div>
        </FormProvider>
      </TypographyProvider>
    );

    return <Example />
  }}
</Playground>


### **Theme with different form component states and common style**

You can style form component states and add common style for each of the form component group.

<Playground>
  {() => {
    const formTheme = {
      regular: {
        states: {
          valid: {
            normal: {
              borderColor: '#ced4da',
              color: '#161616',
            },
            active: {
              borderColor: '#1E88E5',
            },
            disabled: {
              color: '#868e96',
              backgroundColor: '#f8f9fa',
            },
          },
          invalid: {
            color: '#F4511E',
            borderColor: '#F4511E',
          },
        },
      },
      inline: {
        states: {
          valid: {
            normal: {
              color: '#161616',
            },
            hover: {
              borderColor: '#ced4da',
              color: '#161616',
            },
            active: {
              borderColor: '#1E88E5',
              color: '#161616',
            },
            disabled: {
              borderColor: '#ced4da',
              color: '#868e96',
              backgroundColor: '#f8f9fa',
            },
          },
          invalid: {
            color: '#F4511E',
            borderColor: '#F4511E',
          },
        },
      },
      misc: {
        states: {
          valid: {
            normal: {
              borderColor: '#ced4da',
            },
            active: {
              borderColor: '#1E88E5',
            },
            checked: {
              borderColor: '#1E88E5',
              backgroundColor: '#1E88E5',
            },
          },
          invalid: {
            borderColor: '#F4511E',
          },
        },
      },
      toggle: {
        states: {
          normal: {
            backgroundColor: '#ced4da',
          },
          checked: {
            backgroundColor: '#1E88E5',
          },
        },
      },
      common: {
        regular: {
          fontSize: '1rem',
          fontFamily: `'Inter var', sans-serif`,
        },
        inline: {
          fontSize: '1rem',
          fontFamily: `'Inter var', sans-serif`,
        },
        misc: {
          // to change the size of checkbox and radio input you can increase/decrease the font size
          fontSize: '1rem',
        },
      },
    };

    const Example = () => (
      /* TypographyProvider is required for FormProvider to work */
      <TypographyProvider>
        <FormProvider theme={formTheme}>
          <div className="max-width-m">
            <Input invalid={true} className="margin-bottom" />
            <Input disabled={true} value="disabled" className="margin-bottom" />
            <InlineInput invalid={true} placeholder="Inline input" className="margin-bottom"/>
            <InlineInput disabled={true} placeholder="Inline input" className="margin-bottom"/>
            <Textarea invalid={true} className="margin-bottom" />
            <Textarea disabled={true} className="margin-bottom" />
            <Select invalid={true} className="margin-bottom">
              <option>option one</option>
              <option>option two</option>
            </Select>
            <Select disabled={true} className="margin-bottom">
              <option>option one</option>
              <option>option two</option>
            </Select>
            <div className="margin-bottom">
              <Radio invalid={true} />
              <Radio disabled={true} />
            </div>
            <div className="margin-bottom">
              <Checkbox invalid={true} />
              <Checkbox disabled={true} />
            </div>
            <div className="max-width-s">
              <Toggle disabled={true} onChange={() => {}} />
            </div>
          </div>
        </FormProvider>
      </TypographyProvider>
    );

    return <Example />
  }}
</Playground>
