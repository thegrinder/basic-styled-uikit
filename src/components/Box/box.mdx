## **BoxProvider**

`<BoxProvider />` is responsible for passing down the theme to the `<Box />` component. It expects the theme to be of certain structure.


---


## **Box**

Box component grabs the theme given to `<BoxProvider />` and applies the style defined there based on the given `bgColor` and `elevation` prop values.

<Props of={Box} />


---


### **Example**

<Playground>
  {() => {
    const boxTheme = {
      bgColors: {
        neutral: '#fff',
        primary: '#1e88e5',
        success: '#43a047',
        warning: '#fdd835',
        danger: '#f4511e',
      },
      elevations: {
        1: {
          boxShadow:
            '0 3px 1px -2px rgba(0,0,0,.2), 0 2px 2px 0 rgba(0,0,0,.14), 0 1px 5px 0 rgba(0,0,0,.12)',
        },
        2: {
          boxShadow:
            '0 2px 4px -1px rgba(0,0,0,.2), 0 4px 5px 0 rgba(0,0,0,.14), 0 1px 10px 0 rgba(0,0,0,.12)',
        },
      },
    };

    const Example = () => (
      <BoxProvider theme={boxTheme}>
        <>
          <h2>Background colors:</h2>
          <Box bgColor="neutral" className="margin-bottom">neutral</Box>
          <Box bgColor="primary" className="margin-bottom">primary</Box>
          <Box bgColor="danger" className="margin-bottom">danger</Box>
          <Box bgColor="success" className="margin-bottom">success</Box>
          <Box bgColor="warning" className="margin-bottom">warning</Box>

          <h2>Elevations:</h2>
          <Box bgColor="neutral" className="padding">
            <Box elevation={1} className="padding">
              <span>one</span>
              <Box elevation={1} className="padding">
                <span>one</span>
                <Box elevation={1} className="padding">
                  <span>one</span>
                </Box>
              </Box>
            </Box>

            <Box elevation={2}>
              <span>two</span>
            </Box>
          </Box>

        </>
      </BoxProvider>
    );

    return <Example />;
  }}
</Playground>
